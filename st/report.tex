\section{課題２：大数の強法則}
\label{sec:org51f9d16}
\subsection{\(\overline{X}\) の期待値を示せ。また、nが100のとき、20,000のとき、2,000,000の、 \(\overline{X}\) の分散を手計算で求めよ}
\label{sec:orgc190c0f}
\begin{itemize}
\item \(\overline{X}\) の期待値\\
\(E[\overline{X}]
     = E[1/n{\Sigma^{n}_{i=1}{X_k}}] \\
     = 1/n {\Sigma^{n}_{k-1}E[X_k]} \\
     = 1/n {\Sigma^{n}_{k-1}p}\\
     = 1/n (n p) \\
     = p \\
     \because
     E[X_k]
     = \Sigma_{k=0}^{1}kP\{X_k\} \\ 
     = 0 * (1-p) + 1 * p \\
     = p\)\\
\item \(n = 100\) のときの分散\\
\(p(1-p)/n
     = 0.4(1-0.4)/100 \\
     = 0.24/100 \\
     = 0.0024\)\\
\item \(n = 20,000\) のときの分散\\
\(p(1-p)/n
     = 0.4(1-0.4)/20000 \\
     = 0.24/20000 \\
     = 0.000012\)     \\
\item \(n = 20,000,000\) のときの分散\\
\(p(1-p)/n
     = 0.4(1-0.4)/20000000 \\
     = 0.24/20000000 \\
     = 0.000000012\)          \\
\end{itemize}
